Area:

#include <stdio.h>

float area(float radius)
{
    return 3.14f * radius * radius;
}

int main(void)
{
    for(int i = 1; i <= 10; i++) {
        float a = area(i);
        if(a >= 10 && a <= 100) {
            printf("%d -> area %.2f\n", i, a);
        }
    }
}

FizzBuzz:

#include <stdio.h>

int main(void)
{
    for(int i = 1; i <= 100; i++) {
        if(i % 15 == 0) {
            printf("FizzBuzz\n");
        } else if(i % 3 == 0) {
            printf("Fizz\n");
        } else if(i % 5 == 0) {
            printf("Buzz\n");
        } else {
            printf("%d\n", i);
        }
    }
}
