def area(radius):
    return 3.14 * radius * radius

for i in range(1, 11):
    print("radius %d => area %f" % (i, area(i)))

def square(x):
    return x * x

for i in range(1, 11):
    a = area(i)
    if a > 10 and a < 100:
        print("%d %d" % (i, a))

for i in range(1, 11):
    a = area(i)
    s = square(i)
    print(a - s)
